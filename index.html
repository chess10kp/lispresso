(ns lispresso.fib (:require [clojure.core.reducers :as r])) (defn fib [n] (let [memo (atom {0 0, 1 1})] (fn [n] (if-let [res (@memo n)] res (let [res (+ ((@memoize-fib) (dec n)) ((@memoize-fib) (- n 2)))] (swap! memo assoc n res) res))))) (defn parallel-fib [n] (let [tasks (map #(future (fib %)) (range n))] (map deref tasks)))
